29class Attachment:
30    """Attachment Class"""
31    name: str
32    """File name of file if available. Defaults to file"""
33    data : str
34    """Base64 data representation of file"""

36    def __init__(self, name: str=None):

42    def sendFormat(self):
43        """Creates an attachment array"""
44        decoded_bytes = base64.b64decode(self.data)
45
46        # Use python-magic to get the MIME type
47        mime_type = magic.from_buffer(decoded_bytes, mime=True)
48
49        #print(f'MIME type: {mime_type}')
50        extension = mimetypes.guess_extension(mime_type)
51
52        # Print the MIME type and extension
53        #print(f'MIME type: {mime_type}')
54        #print(f'Extension: {extension}')
55
56        try:
57            #print("{}.{}".format(self.name, extension))
58            name = "{}{}".format(self.name, extension)
59        except:
60            #print("file{}".format(extension))
61            name = "file{}".format(extension)
62
63        return {"name":name, "data":self.data}
